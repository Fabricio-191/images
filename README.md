## Use example

```js
const Images = require('@fabricio-191/images');
const images = new Images({
	checkImages: true,
	addDefaultCategories: true,
});

Images.getFromCategory('cat')
	.then(console.log)
	.catch(console.error);

images.getFromSubreddit('nsfw')
	.then(console.log)
	.catch(console.error);

images.search('runescape')
	.then(console.log)
	.catch(console.error);
```
